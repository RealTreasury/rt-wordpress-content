function updateDropdownPositioning() { // Simple function - no complex calculations needed const banner = document.querySelector('.workshop-banner'); const navContainer = document.querySelector('.rt-nav-container'); if (!navContainer) return; // Just handle banner positioning for nav container if (banner && banner.style.display !== 'none') { if (banner.classList.contains('minimized')) { document.body.classList.add('banner-minimized'); } else { document.body.classList.add('banner-present'); } } else { document.body.classList.add('banner-closed'); } } window.addEventListener('load', updateDropdownPositioning); window.addEventListener('resize', updateDropdownPositioning); document.addEventListener('DOMContentLoaded', function() { // Mobile hamburger toggle functionality const mobileToggle = document.querySelector('.rt-mobile-toggle'); const navMenu = document.querySelector('.rt-nav'); if (mobileToggle && navMenu) { mobileToggle.addEventListener('click', function(e) { e.preventDefault(); e.stopPropagation(); // Toggle mobile menu const isMenuActive = navMenu.classList.contains('active'); if (isMenuActive) { // Close menu navMenu.classList.remove('active'); mobileToggle.classList.remove('active'); document.body.classList.remove('menu-open'); } else { // Open menu navMenu.classList.add('active'); mobileToggle.classList.add('active'); document.body.classList.add('menu-open'); // Close any open dropdowns when opening mobile menu document.querySelectorAll('.rt-nav-item.active').forEach(item => { item.classList.remove('active'); }); } }); } // Dropdown functionality (existing code with improvements) const dropdownLinks = document.querySelectorAll('.rt-nav-link.has-dropdown'); dropdownLinks.forEach(link => { link.addEventListener('click', function(e) { e.preventDefault(); e.stopPropagation(); const navItem = this.closest('.rt-nav-item'); const isActive = navItem.classList.contains('active'); // Close all other dropdowns document.querySelectorAll('.rt-nav-item.active').forEach(item => { if (item !== navItem) { item.classList.remove('active'); } }); // Toggle current dropdown if (!isActive) { navItem.classList.add('active'); } else { navItem.classList.remove('active'); } // After toggling active class, position the dropdown if (navItem.classList.contains('active')) { positionDropdown(navItem); } }); }); // Calculate and set dropdown position for full width function positionDropdown(navItem) { const dropdown = navItem.querySelector('.rt-dropdown'); if (!dropdown) return; const navItemRect = navItem.getBoundingClientRect(); const viewportWidth = window.innerWidth; const leftOffset = -navItemRect.left; dropdown.style.setProperty('margin-left', `${leftOffset}px`, 'important'); dropdown.style.setProperty('width', `${viewportWidth}px`, 'important'); } // Recalculate on window resize window.addEventListener('resize', function() { document.querySelectorAll('.rt-nav-item.active').forEach(navItem => { positionDropdown(navItem); }); }); // Close dropdowns and mobile menu when clicking outside document.addEventListener('click', function(e) { // Don't close if clicking inside nav area if (!e.target.closest('.rt-nav-container')) { // Close all dropdowns document.querySelectorAll('.rt-nav-item.active').forEach(item => { item.classList.remove('active'); }); // Close mobile menu on mobile if (window.innerWidth <= 992) { if (navMenu) navMenu.classList.remove('active'); if (mobileToggle) mobileToggle.classList.remove('active'); document.body.classList.remove('menu-open'); } } }); // Handle window resize - close mobile menu if resizing to desktop window.addEventListener('resize', function() { if (window.innerWidth > 992) { if (navMenu) navMenu.classList.remove('active'); if (mobileToggle) mobileToggle.classList.remove('active'); document.body.classList.remove('menu-open'); // Close all dropdowns on resize document.querySelectorAll('.rt-nav-item.active').forEach(item => { item.classList.remove('active'); }); } }); // Prevent menu from closing when clicking inside dropdown content document.querySelectorAll('.rt-dropdown').forEach(dropdown => { dropdown.addEventListener('click', function(e) { e.stopPropagation(); }); }); });
